Angular Components & Templates

This section covers the core concepts of Angular components, templates, inputs, outputs, lifecycle hooks, and content projection.

Component Basics

An Angular component consists of a TypeScript class, an HTML template, and CSS styles.

Example: Basic Component


// basic-example.component.ts
import {{ '{' }} Component } from '@angular/core';

@Component({{ '{' }}
  selector: 'app-basic-example',
  standalone: true,
  imports: [], // CommonModule if using ngIf, ngFor, etc.
  template: \`
    <h3>{{'{'}}{{'{'}} title }}</h3>
    <p>This is a basic Angular component.</p>
  \`,
  styles: [\`h3 { color: steelblue; }\`]
})
export class BasicExampleComponent {{ '{' }}
  title = 'My Basic Component';
}

To use this component, you would add <app-basic-example></app-basic-example> to another component's template, provided BasicExampleComponent is imported if the host is standalone.

Input Properties (@Input())

The @Input() decorator allows a parent component to pass data to a child component.

Example: Child Component with Input


// child-with-input.component.ts
import {{ '{' }} Component, Input } from '@angular/core';

@Component({{ '{' }}
  selector: 'app-child-with-input',
  standalone: true,
  template: \`
    <p>Message from parent: {{'{'}}{{'{'}} message }}</p>
  \`
})
export class ChildWithInputComponent {{ '{' }}
  @Input() message: string = ''; // Default value
}

Output Properties (@Output() and EventEmitter)

The @Output() decorator allows a child component to emit events up to its parent component. It's used in conjunction with EventEmitter.

Example: Child Component Emitting an Event


// child-emitter.component.ts
import {{ '{' }} Component, Output, EventEmitter } from '@angular/core';

@Component({{ '{' }}
  selector: 'app-child-emitter',
  standalone: true,
  template: \`
    <button (click)="notifyParent()">Notify Parent</button>
  \`
})
export class ChildEmitterComponent {{ '{' }}
  @Output() customEvent = new EventEmitter<string>();

  notifyParent() {{ '{' }}
    this.customEvent.emit('Hello from child!');
  }
}

Component Lifecycle Hooks

Angular components have a lifecycle managed by Angular. You can tap into key moments in this lifecycle by implementing lifecycle hook methods.

Common Lifecycle Hooks:

  • ngOnInit(): Called once, after the first ngOnChanges(). Used for component initialization, data fetching.
  • ngOnChanges(): Called before ngOnInit() and whenever one or more data-bound input properties change. Receives a SimpleChanges object.
  • ngOnDestroy(): Called just before Angular destroys the component. Used for cleanup, unsubscribing from observables.

Example: Using Lifecycle Hooks


// lifecycle-example.component.ts
import {{ '{' }} Component, OnInit, OnChanges, OnDestroy, Input, SimpleChanges } from '@angular/core';

@Component({{ '{' }}
  selector: 'app-lifecycle-example',
  standalone: true,
  template: \`<p>Lifecycle example. Check console. Input: {{'{'}}{{'{'}} myInput }}</p>\`
})
export class LifecycleExampleComponent implements OnInit, OnChanges, OnDestroy {{ '{' }}
  @Input() myInput: string = '';
  private intervalId: any;

  constructor() {{ '{' }}
    console.log('Lifecycle: Constructor called');
  }

  ngOnChanges(changes: SimpleChanges) {{ '{' }}
    console.log('Lifecycle: ngOnChanges called', changes);
  }

  ngOnInit() {{ '{' }}
    console.log('Lifecycle: ngOnInit called');
    this.intervalId = setInterval(() => {{ '{' }}
      console.log('Lifecycle: Interval running');
    }, 2000);
  }

  ngOnDestroy() {{ '{' }}
    console.log('Lifecycle: ngOnDestroy called');
    if (this.intervalId) {{ '{' }}
      clearInterval(this.intervalId);
    }
  }
}

Content Projection (ng-content)

Content projection allows you to create components that can host content passed in from a parent component. This is achieved using the <ng-content> element.

Example: Basic Content Projection


// content-wrapper.component.ts
import {{ '{' }} Component } from '@angular/core';

@Component({{ '{' }}
  selector: 'app-content-wrapper',
  standalone: true,
  template: \`
    <div style="border: 1px solid #ccc; padding: 10px;">
      <h4>Wrapper Title</h4>
      <ng-content></ng-content> <!-- Content from parent will be projected here -->
      <p>Wrapper Footer</p>
    </div>
  \`
})
export class ContentWrapperComponent {{ '{' }} }

Multi-slot Content Projection (with select attribute)


// multi-slot-wrapper.component.ts
import {{ '{' }} Component } from '@angular/core';

@Component({{ '{' }}
  selector: 'app-multi-slot-wrapper',
  standalone: true,
  template: \`
    <div style="border: 1px solid lightcoral; padding: 10px;">
      <h4>Multi-Slot Wrapper</h4>
      <div class="header">
        <ng-content select="[slot=header]"></ng-content> <!-- Project content with slot='header' attribute -->
      </div>
      <hr>
      <ng-content></ng-content> <!-- Default slot for content without a slot attribute -->
      <hr>
      <div class="footer">
        <ng-content select=".footer-content"></ng-content> <!-- Project content with class 'footer-content' -->
      </div>
    </div>
  \`
})
export class MultiSlotWrapperComponent {{ '{' }} }
